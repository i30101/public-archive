Question #0110


    ALPHABET = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"

    # mystery cipher #2
    def mystery2(plain: str, keyword: str):
        # obtain letters from plaintext
        plaintext = "".join([letter for letter in plain if letter in ALPHABET])
            
        # split alphabet
        first_half = ALPHABET[:13]
        second_half = ALPHABET[13:]
        
        rows = [second_half[i:] + second_half[:i] for i in range(13)]
        
        # encryption
        ciphertext = ""
        for j, plain_letter in enumerate(plaintext):
            row_index = int(ALPHABET.index(keyword[j % len(keyword)]) / 2)
            if plain_letter in first_half:
                ciphertext += rows[row_index][first_half.index(plain_letter)]
            else:
                ciphertext += ALPHABET[rows[row_index].index(plain_letter)]
        
        return ciphertext
